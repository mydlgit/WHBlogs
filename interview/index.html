  • 将阿拉伯数字转成大写中文，数字整数部分不会超过九位，小数部分不超过三位
void test(double _money){
    vector<string> nums{
        "零", "壹", "贰", "叁", "肆", "伍", "陆", "柒", "捌", "玖"
    };
    vector<string> radix{
        "圆", "拾", "佰", "仟", "万", "拾", "佰", "仟", "亿"
    };
    int money = (int)_money, point = (int)(_money - money) * 1000 + 1;
    vector<string> ans;
    int cnt = 0, tmp=0;
    while (money) {
        tmp = money % 10;
        if (tmp != 0) {
            ans.emplace_back(radix[cnt]);
            ans.emplace_back(nums[tmp]);
        }
        else {
            if ((cnt % 4)==0)ans.emplace_back(radix[cnt]);
            else if (ans.back() != "零" && ans.back()!="圆"  && ans.back() != "万") {
                ans.emplace_back("零");
            }
        }
        cnt++;
        money /= 10;
    }
    string result = "";
    for (int i = ans.size() - 1;i >= 0;i--) {
        result += ans[i];
    }
    vector<string> extra;
    cnt = 1;
    while (point) {
        tmp = point%10;
        point /= 10;
        if (tmp != 0) {
            if (cnt == 1)extra.emplace_back("厘");
            else if (cnt == 2)extra.emplace_back("分");
            else if (cnt == 3) extra.emplace_back("角");
            extra.emplace_back(nums[tmp]);
        }
        cnt++;
    }
    for(int i=extra.size()-1;i>=0;i--)
        result += extra[i];
    cout << result << endl;
}

  • 共有n台服务器，每台服务器每秒处理的最大请求数为qps[ i ], 设计一个调度算法，每个客服端使用该调度算法时会生成一个服务器编号(0~n-1)，设计的算法应保证同一时刻有多个客户端发起请求时，某个服务器不会过载，或者说server[ i ]接收的请求数与server[ j ]接收的请求数应有server[ i ]:server[ j ] = qps[ i ]:qps[ j ].
void findX(vector<int>& qps){
    int sum=0;
    int len=qps.size();
    vector<int> vec(len+1, 0);
    for(int i=0;i<len;i++){
        sum += qps[i];
        vec[i+1] = sum;
    }
    srand(clock());
    int target=rand()%(sum-1);
    int left=0, right=len;
    cout<<"target:"<<target<<endl;  //模拟每次客户端每次发送请求时产生的随机数
    while(left<=right){
        int mid=(left+right)/2;
        if(vec[mid] < target)left=mid+1;
        else right=mid-1;
    }
    if(vec[left]==target)cout<<left;
    else cout<<left-1;
    cout<<endl;
}
int main(){
    vector<int> qps{3, 6, 9, 2, 10, 5};
    findX(qps);
    return 0;
}
  • 有n个排序好的数组/链表，将它们合并成一个有序的数组/链表
/*
给你一个链表数组，每个链表都已经按升序排列。请你将所有链表合并到一个升序链表中，返回合并后的链表.
lists中可能存在空链表
采用堆的方式
*/
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        if(lists.size()==0)return nullptr;
        int k=lists.size(), cnt=k;
        priority_queue<int, vector<int>, greater<int> >  heap;
        ListNode *root = new ListNode(-1), *cur=root;
        while(cnt){
            cnt=k;
            for(auto& node:lists){
                if(node){
                    heap.push(node->val);
                    node=node->next;
                }else cnt--;
            }
            if(cnt==0)break;
            cur->next=new ListNode(heap.top());
            heap.pop();
            cur = cur->next;
        }
        while(!heap.empty()){
            cur->next=new ListNode(heap.top());
            heap.pop();
            cur = cur->next;
        }
        return root->next;
    }
};

  • hashmap的数组大小为什么是2的幂次
    h=hash(key), h在数组中的位置为 h%length， 而如果length是2的整数次幂，则 h%length 等于 h&(length-1)，而&的效率要高于%，出于这个原因，数组扩容时直接翻倍即可。

  • 给定一个数组，给定一个数字，返回数组中所有两数之和为指定数字的两个数的索引 

void findIndex(vector<int>& nums, int target, vector<vector<int>>& ans){
    unordered_map<int, vector<int>> memo;
    for(int i=0; i<nums.size(); i++){
        if(memo.count(nums[i]) == 0){
            memo[nums[i]] = vector<int>{i};
        }else{
            memo[nums[i]].push_back(i);
        }
    }
    for(auto &v:memo){
        if(memo.count(target - v.first)){
            for(auto& a1:v.second){
                vector<int> res(2);
                res[0]=a1;
                for(auto& a2:memo[target-v.first]){
                    res[1] = a2;
                    ans.push_back(res);
                }
            }
            memo.erase(target-v.first);
        }
    }
}
int main()
{   
    vector<int> nums{4, 4, 2, 4};
    vector<vector<int>> ans;
    findIndex(nums, 6, ans);
    for(auto &v:ans){
        for(int i:v){
            cout<<i<<" ";
        }
        cout<<endl;
    }
    return 0;
}
